---
title: 'Create Template'
api: 'POST https://api.vizapi.ai/v1/templates/create'
description: 'Create a new template for document extraction'
---

The `/templates/create` endpoint allows you to create a new template for document extraction.

## Request

<ParamField header="x-api-key" type="string" required>
  Your API key for authentication.
</ParamField>

<ParamField body="name" type="string" required>
  The name of the template.
</ParamField>

<ParamField body="description" type="string" required>
  A description of the template.
</ParamField>

<ParamField body="category" type="string" required>
  The category of the template (e.g., "invoice", "receipt", "id_card").
</ParamField>

<ParamField body="tags" type="array">
  An array of tags associated with the template.
</ParamField>

<ParamField body="fields" type="object" required>
  The field definitions for the template.
</ParamField>

<ParamField body="metadata" type="object">
  Additional metadata for the template.
</ParamField>

<ParamField body="is_public" type="boolean">
  Whether the template is public or private. Default is `false`.
</ParamField>

<ParamField body="user_id" type="string" required>
  The ID of the user creating the template.
</ParamField>

## Response

<ResponseField name="id" type="string">
  The unique identifier for the created template.
</ResponseField>

<ResponseField name="name" type="string">
  The name of the template.
</ResponseField>

<ResponseField name="description" type="string">
  The description of the template.
</ResponseField>

<ResponseField name="category" type="string">
  The category of the template.
</ResponseField>

<ResponseField name="tags" type="array">
  The tags associated with the template.
</ResponseField>

<ResponseField name="popularity" type="string">
  The popularity level of the template ("high", "medium", "low").
</ResponseField>

<ResponseField name="usage_count" type="number">
  The number of times the template has been used.
</ResponseField>

<ResponseField name="last_updated" type="string">
  The date when the template was last updated.
</ResponseField>

<ResponseField name="added_date" type="string">
  The date when the template was added.
</ResponseField>

<ResponseField name="icon" type="string">
  The icon associated with the template.
</ResponseField>

<ResponseField name="job_id" type="string">
  The job ID associated with the template.
</ResponseField>

<ResponseField name="created_at" type="string">
  The timestamp when the template was created.
</ResponseField>

<ResponseField name="updated_at" type="string">
  The timestamp when the template was last updated.
</ResponseField>

<ResponseField name="fields" type="object">
  The field definitions for the template.
</ResponseField>

<ResponseField name="is_public" type="boolean">
  Whether the template is public or private.
</ResponseField>

<ResponseField name="metadata" type="object">
  Additional metadata for the template.
</ResponseField>

```javascript Node.js
const fetch = require('node-fetch');

async function createTemplate() {
  const response = await fetch('https://api.vizapi.ai/v1/templates/create', {
    method: 'POST',
    headers: {
      'x-api-key': 'YOUR_API_KEY',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify({
      name: 'Invoice Template',
      description: 'Template for extracting data from invoices',
      category: 'invoice',
      tags: ['finance', 'accounting'],
      fields: {
        invoice_number: {
          type: 'string',
          description: 'The invoice identifier'
        },
        date: {
          type: 'string',
          description: 'The invoice date'
        },
        total: {
          type: 'number',
          description: 'The total amount'
        }
      },
      metadata: {
        version: '1.0'
      },
      is_public: false,
      user_id: 'user_12345'
    })
  });
  
  const data = await response.json();
  console.log(data);
}

createTemplate();
```

```python Python
import requests

url = "https://api.vizapi.ai/v1/templates/create"
headers = {
    "x-api-key": "YOUR_API_KEY",
    "Content-Type": "application/json"
}
payload = {
    "name": "Invoice Template",
    "description": "Template for extracting data from invoices",
    "category": "invoice",
    "tags": ["finance", "accounting"],
    "fields": {
        "invoice_number": {
            "type": "string",
            "description": "The invoice identifier"
        },
        "date": {
            "type": "string",
            "description": "The invoice date"
        },
        "total": {
            "type": "number",
            "description": "The total amount"
        }
    },
    "metadata": {
        "version": "1.0"
    },
    "is_public": False,
    "user_id": "user_12345"
}

response = requests.post(url, headers=headers, json=payload)
data = response.json()
print(data)
```
