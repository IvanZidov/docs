---
title: 'Get Public Templates'
api: 'GET https://api.vizapi.ai/v1/templates/public'
description: 'Get all public templates'
---

The `/templates/public` endpoint retrieves all public templates available in the system.

## Request

<ParamField header="x-api-key" type="string" required>
  Your API key for authentication.
</ParamField>

## Response

<ResponseField name="templates" type="array">
  An array of template objects.
  
  <Expandable title="Template object">
    <ResponseField name="id" type="string">
      The unique identifier for the template.
    </ResponseField>
    
    <ResponseField name="name" type="string">
      The name of the template.
    </ResponseField>
    
    <ResponseField name="description" type="string">
      The description of the template.
    </ResponseField>
    
    <ResponseField name="category" type="string">
      The category of the template.
    </ResponseField>
    
    <ResponseField name="tags" type="array">
      The tags associated with the template.
    </ResponseField>
    
    <ResponseField name="popularity" type="string">
      The popularity level of the template ("high", "medium", "low").
    </ResponseField>
    
    <ResponseField name="usage_count" type="number">
      The number of times the template has been used.
    </ResponseField>
    
    <ResponseField name="last_updated" type="string">
      The date when the template was last updated.
    </ResponseField>
    
    <ResponseField name="added_date" type="string">
      The date when the template was added.
    </ResponseField>
    
    <ResponseField name="icon" type="string">
      The icon associated with the template.
    </ResponseField>
    
    <ResponseField name="job_id" type="string">
      The job ID associated with the template.
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      The timestamp when the template was created.
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      The timestamp when the template was last updated.
    </ResponseField>
    
    <ResponseField name="fields" type="object">
      The field definitions for the template.
    </ResponseField>
    
    <ResponseField name="is_public" type="boolean">
      Whether the template is public (always true for this endpoint).
    </ResponseField>
    
    <ResponseField name="metadata" type="object">
      Additional metadata for the template.
    </ResponseField>
  </Expandable>
</ResponseField>

## Examples

### Example Request

<CodeGroup>

```bash cURL
curl -X GET https://api.vizapi.ai/v1/templates/public \
  -H "x-api-key: YOUR_API_KEY"
```

```javascript Node.js
const fetch = require('node-fetch');

async function getPublicTemplates() {
  const response = await fetch('https://api.vizapi.ai/v1/templates/public', {
    method: 'GET',
    headers: {
      'x-api-key': 'YOUR_API_KEY'
    }
  });
  
  const data = await response.json();
  console.log(data);
}

getPublicTemplates();
```

```python Python
import requests

url = "https://api.vizapi.ai/v1/templates/public"
headers = {
    "x-api-key": "YOUR_API_KEY"
}

response = requests.get(url, headers=headers)
data = response.json()
print(data)
```

</CodeGroup>
